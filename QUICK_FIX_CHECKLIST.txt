□ FIX-4: SSE write() 添加错误处理
  文件: web/server/routes/analysis.ts
  位置: L92-93
  改动:
    // 改这个：
    const send = (event: ProgressEvent) => {
      reply.raw.write(`data: ${JSON.stringify(event)}\n\n`);
    };
    
    // 改为这个：
    const send = (event: ProgressEvent) => {
      try {
        if (!reply.raw.writable) {
          console.warn(`SSE connection closed for ${id}`);
          return;
        }
        
        const canContinue = reply.raw.write(
          `data: ${JSON.stringify(event)}\n\n`
        );
        
        if (!canContinue) {
          console.log(`SSE backpressure for ${id}`);
          // 可选：暂停 CLI 输出
        }
      } catch (err) {
        console.error(`SSE write error for ${id}:`, err);
        // 移除该客户端
        const clients = sseClients.get(id);
        if (clients) {
          clients.delete(send);
        }
      }
    };
  
  验证: send 函数有 try-catch 和检查
